    interval=1

    counter_dir="/sys/class/infiniband/$ib_dev/ports/1/counters"

    old_recv_packets[$ib_dev]=$(cat $counter_dir/port_rcv_packets)
    old_xmit_bytes[$ib_dev]=$(cat $counter_dir/port_xmit_data)

        counter_dir="/sys/class/infiniband/$ib_dev/ports/1/counters"
